async function submitToGoogle() { const btn = document.getElementById('saveBtn'); const nameInput = document.getElementById('b_name').value; if(!nameInput) return alert("請填寫姓名！"); btn.disabled = true; btn.innerText = "📍 正在強力上傳..."; // 重新封裝資料為參數格式 const m = Array.from({length:5}, (_,i) => parseInt(document.getElementById('dm_'+i)?.value || 0)).reduce((a,b)=>a+b, 0); const sc = (m >= 15 ? 3 : (m >= 10 ? 2 : (m >= 5 ? 1 : 0))) + parseInt(document.getElementById('c_time').value); const getV = (id) => (document.getElementById(id).innerText.match(/\d/) || [0])[0]; // 建立參數包 const params = new URLSearchParams(); params.append('name', nameInput); params.append('gender', document.getElementById('b_gender').value); params.append('age', document.getElementById('b_age').value); params.append('weighted_total', document.getElementById('rt').innerText); params.append('cognition_score', sc); params.append('physical_score', parseInt(getV('r_chair')) + parseInt(getV('r_arm')) + parseInt(getV('s_gait'))); params.append('life_score', document.getElementById('l_med').value); params.append('nutrition_score', document.getElementById('n_veg').value); params.append('social_score', parseInt(document.getElementById('ps_1').value) + parseInt(document.getElementById('ps_2').value) + parseInt(document.getElementById('ps_3').value)); params.append('chair_count', document.getElementById('p_chair').value); params.append('arm_count', document.getElementById('p_arm').value); params.append('gait_time', document.getElementById('p_gait').value); params.append('status', document.getElementById('rs').innerText); try { // 使用 URLSearchParams 傳送，這是對 Google Sheets 最穩定的方法 await fetch(scriptURL, { method: 'POST', mode: 'no-cors', headers: { 'Content-Type': 'application/x-www-form-urlencoded' }, body: params.toString() }); alert("✅ 澎邱完整版：資料已成功寫入試算表！"); btn.innerText = "已存檔"; } catch(e) { alert("連線失敗，請確認 Script 網址是否為最新版本。"); btn.disabled = false; btn.innerText = "重新儲存"; } }
